Extension Lambdas
An extension lambda is like an extension function, except it defines a
lambda instead of a function.
Here,
va
and
vb
yield the same result:

import
atomictest.eq
val
va: (String, Int) -> String = { str, n ->
str.repeat(n) + str.repeat(n)
}
val
vb: String.(Int) -> String = {
this
.repeat(it) + repeat(it)
}

main
() {
va(
"Vanbo"

2
) eq

"Vanbo"
.vb(
2
) eq

vb(
"Vanbo"
,
2
) eq
"VanboVanboVanboVanbo"
// "Vanbo".va(2) // Doesn't compile
}

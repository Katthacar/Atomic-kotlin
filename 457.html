resourcecleanup
import
atomictest.*
class
Usable
() : AutoCloseable {
fun
func
() = trace(
"func()"
)
override fun
close
() = trace(
"close()"
)
}
fun
main
() {
Usable().use { it.func() }
trace eq
"""
func()
close()
"""
}
use()
ensures resource cleanup at the point the resource is created, instead of forcing
you to write code when you’re finished with it.

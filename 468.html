package
unittesting
import
unittesting.State.*
enum class
State
{ On, Off, Paused }
class
StateMachine
{
var
state: State = Off
private set
private fun
transition
(
new: State, current: State = On
){
if
(new == Off && state != Off)
state = Off
else if
(state == current)
state = new
}
fun
start
() = transition(On, Off)
fun
pause
() = transition(Paused, On)
fun
resume
() = transition(On, Paused)
fun
finish
() = transition(Off)
}
These operations are ignored:
•
resume()
or
finish()
on a machine that is
Off
.
•
pause()
or
start()
on a
Paused
machine.

StateMachine
, we create a property
sm

StateMachineTest
object before running each test, so you start with
a fresh
StateMachine
object for each different test:

Iterating over a
List
with index uses the same destructuring feature:
// Destructuring/LoopWithIndex.kt
import
atomictest.trace
fun
main
() {
val
list = listOf(
'a'
,
'b'
,
'c'
)
for
((index, value)
in
list.withIndex()) {
trace(
"$index:$value"
)
}
trace eq
"0:a 1:b 2:c"
}
